CREATE UNIQUE INDEX PERSON_ID_INDEX ON PERSON (
    library_id COLLATE BINARY ASC
);

CREATE UNIQUE INDEX MEDIA_ITEM_INDEX ON MEDIA_ITEM (
    doi_eidr COLLATE BINARY ASC
);

    inventory_number COLLATE BINARY,
    library_id COLLATE BINARY,
    checkout_date COLLATE BINARY
);

CREATE INDEX COPY_INDEX ON COPY (
    inventory_number COLLATE BINARY,
    doi_eidr COLLATE BINARY
    location COLLATE BINARY
);
